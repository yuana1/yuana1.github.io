1 事件简介

java事件是awt/swing编程中必不可少的一部分。它涉及了三个参与者

  1. 事件 事件对象，例如单击事件，键盘按下事件等等。java中所有的事件对象都继承自EventObject对象，它内部包含了一个事件源对象。
  2. 事件源 产生事件的对象，例如按钮，文本框等等。事件源上可以注册监听器，监听事件的发生并处理事件。
  3. 监听器 当对象发生之后需要执行动作的对象。java中所有的监听器都实现了EventListener接口。该接口只是一个标记接口，标记了他是一个监听器。

事件消息的传递过程，拿最简单的按钮上鼠标点击事件来说

  1. 鼠标点击按钮，产生一个单击事件。
  2. 该事件被系统捕获，将事件放到EventQueue队列中。通过EventDispatchThread线程派发给各个监听器。
  3. 监听器收到之后处理该事件。

2 swing中的事件实现

下面的代码模拟了一个button。包括了单击双击事件，事件监听器，button事件源，EventQueue事件队列，EventDispatchThread派发线程。

代码和swing中的类似，精简了swing中部分代码。 

package springboot.boot;

import java.util.EventListener;
import java.util.EventObject;
import java.util.concurrent.LinkedBlockingQueue;

public class ButtonApp {

    public static void main(String[] args) {

        Button b = new Button();
        b.name = "asdf";
        // 添加一个事件监听器
        b.addMouseListener(new MouseListener() {

            public void dbClick(MouseEvent e) {
                // TODO Auto-generated method stub
                System.out.println("dbclicked!");
            }

            public void click(MouseEvent e) {
                // TODO Auto-generated method stub
                System.out.println("clicked!");
            }
        });
        // 再添加一个监听器
        b.addMouseListener(new MouseListener() {

            public void dbClick(MouseEvent e) {
                // TODO Auto-generated method stub
                System.out.println("2 dbclicked!");
            }

            public void click(MouseEvent e) {
                // TODO Auto-generated method stub
                System.out.println("2 clicked!");
            }
        });
        // b.processEvent(new MouseEvent(b, 1));
        // 创建一个事件队列。
        EventQueue queue = new EventQueue();
        try {
            // 模拟单击事件
            queue.put(new MouseEvent(b, 1));
            // 模拟双击事件
            queue.put(new MouseEvent(b, 2));
            queue.put(new MouseEvent(b, 2));
        } catch (InterruptedException e1) {
            // TODO Auto-generated catch block
            e1.printStackTrace();
        }

    }
}

/**
 * 事件源，继承自Component
 *
 * @author zhaokaiyuan
 *
 */
class Button extends Component {
    String name;

}

/**
 * 一个基类 事件源
 *
 * @author zhaokaiyuan
 *
 */
class Component {

    MouseListener mouseListener; // 鼠标事件监听器

    // 注册鼠标事件监听器
    void addMouseListener(MouseListener ml) {
        mouseListener = MultiListener.add(mouseListener, ml);
    }

    // 根据事件类型处理事件
    void processEvent(CusEvent e) {
        if (e instanceof MouseEvent) {
            switch (e.id) {
            case 1:
                mouseListener.click((MouseEvent) e);
                break;
            case 2:
                mouseListener.dbClick((MouseEvent) e);
                break;
            }
        }

    }
}

/**
 * 添加注册事件时用的。
 *
 * @author zhaokaiyuan
 *
 */
class MultiListener implements MouseListener, EventListener {

    EventListener a, b;

    public MultiListener(EventListener a, EventListener b) {
        // TODO Auto-generated constructor stub
        this.a = a;
        this.b = b;
    }

    public static MouseListener add(MouseListener a, MouseListener b) {
        if (a == null)
            return b;
        if (b == null)
            return a;
        return new MultiListener(a, b);
    }

    public void click(MouseEvent e) {
        // TODO Auto-generated method stub
        ((MouseListener) a).click(e);
        ((MouseListener) b).click(e);

    }

    public void dbClick(MouseEvent e) {
        // TODO Auto-generated method stub
        ((MouseListener) a).dbClick(e);
        ((MouseListener) b).dbClick(e);
    }

}

/**
 * 一个自定义事件基类
 *
 * @author zhaokaiyuan

 */
class CusEvent extends EventObject {

    /**
         *
         */
    private static final long serialVersionUID = 5581196563344575983L;
    int id; // 事件id

    public CusEvent(Object source) {
        super(source);
        // TODO Auto-generated constructor stub
    }

    public CusEvent(Object source, int id) {
        super(source);
        // TODO Auto-generated constructor stub
        this.id = id;
    }

}

/**
 * 鼠标事件
 *
 * @author zhaokaiyuan
 *
 */
class MouseEvent extends CusEvent {

    /**
         *
         */
    private static final long serialVersionUID = 6753428882113432838L;

    public MouseEvent(Object source) {
        super(source);
        // TODO Auto-generated constructor stub
    }

    public MouseEvent(Object source, int id) {
        super(source);
        // TODO Auto-generated constructor stub
        this.id = id;
    }

}

/**
 * 事件监听器
 *
 * @author zhaokaiyuan
 *
 */
interface MouseListener extends EventListener {

    public void click(MouseEvent e);

    public void dbClick(MouseEvent e);

}

/**
 * 事件队列 继承自LinkedBlockingQueue，阻塞队列
 *
 * @author zhaokaiyuan
 *
 */
class EventQueue extends LinkedBlockingQueue {

    EventDT edt; // 分发线程

    public EventQueue() {
        // TODO Auto-generated constructor stub
        edt = new EventDT(this);
        Thread t = new Thread(edt);
        t.setDaemon(false);
        t.start();
    }

    // 派发事件
    public void dispatchEvent(CusEvent e) {

        Component o = (Component) e.getSource();
        o.processEvent(e);
    }

}

/**
 * 派发线程，不断的从队列中取事件，并处理。
 *
 * @author zhaokaiyuan
 *
 */
class EventDT implements Runnable {

    // 事件队列
    public EventQueue queue;

    public EventDT(EventQueue queue) {
        // TODO Auto-generated constructor stub
        this.queue = queue;
    }

    public void run() {
        // TODO Auto-generated method stub
        while (true) {

            try {
                CusEvent o = (CusEvent) queue.take();
                this.queue.dispatchEvent(o);
            } catch (InterruptedException e) {
                // TODO Auto-generated catch block
                e.printStackTrace();
            }

        }
    }
}

3 自定义事件实现

了解了上述代码之后，自定义事件也就非常好写了。只要继承EventObject对象，实现EventListener接口，再编写一个事件源即可。触发事件可以用 事件队列+派发线程事件，也可以手动触发。 手动触发代码只需将创建队列的代码改为如下即可 

b.processEvent(new MouseEvent(b,1));

另外，Component类也可以这样写，这样就不需要MultiListener了 

class Component {
    HashSet<MouseListener> mouseSet = new HashSet<MouseListener>();

    // 注册鼠标事件监听器
    void addMouseListener(MouseListener ml) {
        mouseSet.add(ml);
    }

    // 根据事件类型处理事件
    void processEvent(CusEvent e) {

        if (e instanceof MouseEvent) {
            for(MouseListener mouseListener : mouseSet) {
                switch (e.id) {
                case 1:
                    mouseListener.click((MouseEvent) e);
                    break;
                case 2:
                    mouseListener.dbClick((MouseEvent) e);
                    break;
                }
            }
        }

    }
}
